## Custom Components

### ColorPalette Component

```tsx
// src/components/ColorPalette/index.tsx
import React from 'react';
import styles from './styles.module.css';

interface ColorSwatchProps {
  name: string;
  hex: string;
  usage?: string;
}

export function ColorPalette({ colors }: { colors: ColorSwatchProps[] }) {
  const copyToClipboard = (text: string) => {
    navigator.clipboard.writeText(text);
  };

  return (
    <div className={styles.palette}>
      {colors.map((color) => (
        <div key={color.name} className={styles.swatch}>
          <div
            className={styles.color}
            style={{ backgroundColor: color.hex }}
            onClick={() => copyToClipboard(color.hex)}
          />
          <div className={styles.info}>
            <strong>{color.name}</strong>
            <code>{color.hex}</code>
            {color.usage && <p className={styles.usage}>{color.usage}</p>}
          </div>
        </div>
      ))}
    </div>
  );
}
```

### ComponentPlayground Component

```tsx
// src/components/ComponentPlayground/index.tsx
import React, { useState } from 'react';
import { LiveProvider, LiveEditor, LiveError, LivePreview } from 'react-live';
import styles from './styles.module.css';

export function ComponentPlayground({
  code,
  scope,
  noInline = false,
}: {
  code: string;
  scope: Record<string, any>;
  noInline?: boolean;
}) {
  return (
    <LiveProvider code={code} scope={scope} noInline={noInline}>
      <div className={styles.playground}>
        <div className={styles.preview}>
          <LivePreview />
          <LiveError />
        </div>
        <div className={styles.editor}>
          <LiveEditor />
        </div>
      </div>
    </LiveProvider>
  );
}
```

## Dynamic Subdomain Strategy

For user-specific design systems:

**URL Pattern:** `[brand-id].design.machups.com`

**Implementation:**
1. Vercel wildcard domain: `*.design.machups.com`
2. Middleware extracts brand ID from subdomain
3. Fetch brand data from database/storage
4. Inject brand-specific content into Docusaurus

```typescript
// middleware.ts
export function middleware(req: NextRequest) {
  const hostname = req.headers.get('host');
  const brandId = hostname?.split('.')[0];

  if (brandId && brandId !== 'design') {
    // Fetch brand-specific data
    const brandData = await fetchBrandData(brandId);

    // Rewrite to dynamic page
    return NextResponse.rewrite(
      new URL(`/brand/${brandId}`, req.url)
    );
  }

  return NextResponse.next();
}
```
